רקס פורטונטוס ושות'

תרגום : הילה אהרון בריק

היֹה הָיְתָה פַּעַם עַלְמָה שֶׁמָּצְאָה אֶת עַצְמָהּ בְּגִיל הָעֲמִידָה עִם הַכְנָסָה זְעוּמָה בִּלְבַד. הִיא יָדְעָה שֶׁאֵינָהּ מַשְׂכִּילָה דַּיָּהּ לַעֲסֹק בְּמִשְׁלַח יָד אַחֵר, כָּךְ שֶׁהֶחְלִיטָה, בְּלִי הִסּוּס, לְהִכָּנֵס לִתְחוּם הַהוֹרָאָה. הִיא פָּתְחָה פְּנִימִיָּה מֻבְחֶרֶת לִגְבָרוֹת צְעִירוֹת. לַפְּנִימִיָּה הָיוּ הַמְלָצוֹת מְעֻלּוֹת לְלֹא רְבָב, וְכָאֵלֶּה נִדְרְשׁוּ גַּם מֵהַתַּלְמִידוֹת. וּכְדֵי לְהַשְׁגִּיחַ שֶׁבֵּית הַסֵּפֶר יִהְיֶה מֻבְחָר כְּכָל הָאֶפְשָׁר, קָבְעָה הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן עַל הַדֶּלֶת שֶׁלֶט פְּלִיז, עִם כִּתּוּב בְּאוֹתִיּוֹת קְטַנּוֹת וּמְנֻמָּסוֹת (שְׁמַעְתֶּם בְּוַדַּאי עַל "מִכְתְּבֵי  נִימוּסִין". וּבְכֵן, זֶה מָה שֶׁהָיָה חָרוּת עַל שֶׁלֶט הַדֶּלֶת שֶׁל הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן):

"פְּנִימִיָּה מֻבְחֶרֶת לִבְנוֹת-מְלָכִים מְכֻבָּדִים"

מְלָכִים רַבִּים שֶׁלֹּא הָיוּ מְכֻבָּדִים כְּלָל הָיוּ מוֹסְרִים בְּשִׂמְחָה אֶת אָזְנֵיהֶם הַמַּלְכוּתִיּוֹת, וּבִלְבַד שֶׁיֻּרְשֶׁה לָהֶם לִשְׁלֹחַ אֶת בְּנוֹתֵיהֶם לְבֵית הַסֵּפֶר הַזֶּה, אַךְ הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן הָיְתָה נֻקְשָׁה מְאוֹד בַּאֲשֶׁר לַהַמְלָצוֹת, וְהַתּוֹצָאָה הָיְתָה שֶׁרַק מְלָכִים בְּנֵי הַמַּעֲמָד הַגָּבוֹהַּ בְּיוֹתֵר זָכוּ וְהֻרְשׁוּ לְשַׁלֵּם עֲשֶׂרֶת אֲלָפִים לִירוֹת שְׂטֶרְלִינְג לְשָׁנָה בַּעֲבוּר חִנּוּכָהּ שֶׁל בִּתָּם. וְכָךְ יָכְלָה הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן לָשִׂים בַּצַּד מְעַט כֶּסֶף שֶׁיַּבְטִיחַ אֶת הַכְנָסָתָהּ כְּשֶׁתִּזְדַּקֵּן. וְכָל הַכֶּסֶף שֶׁחָסְכָה הֻשְׁקַע בְּקַרְקָעוֹת.

רַק מֶלֶךְ אֶחָד סֵרֵב לִשְׁלֹחַ אֶת בִּתּוֹ לָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן, עַל בְּסִיס הַטַּעֲנָה שֶׁבֵּית סֵפֶר זוֹל כָּל כָּךְ לֹא יָכוֹל לִהְיוֹת מֻבְחָר בֶּאֱמֶת, אֲבָל כַּעֲבֹר זְמַן נִמְצָא שֶׁהַמְלָצוֹתָיו שֶׁל אוֹתוֹ מֶלֶךְ לֹא הָיוּ מַשְׂבִּיעוֹת רָצוֹן כְּלָל וְעִקָּר.

הָיוּ רַק שֵׁשׁ תַּלְמִידוֹת מַלְכוּתִיּוֹת בַּפְּנִימִיָּה, וּמוּבָן שֶׁהַמּוֹרִים הַמְּעֻלִּים בְּיוֹתֵר הָעָסְקוּ כְּדֵי לְלַמֵּד אוֹתָן כָּל דָּבָר שֶׁהוֹרֵיהֶן רָצוּ שֶׁיִּלְמְדוּ, וּמִכֵּיוָן שֶׁהַבָּנוֹת מֵעוֹלָם לֹא הִתְבַּקְּשׁוּ לַעֲשׂוֹת שִׁעוּרֵי בַּיִת, אֶלָּא אִם כֵּן הֵן גִּלּוּ נְטִיָּה לְכָךְ, הִסְכִּימוּ כֻּלָּן שֶׁזֶּה בֵּית הַסֵּפֶר הַנֶּחְמָד בְּיוֹתֵר בָּעוֹלָם כֻּלּוֹ, וּבָכוּ לְעֶצֶם הַמַּחְשָׁבָה שֶׁיֻּרְחֲקוּ מִמֶּנּוּ. וְכָךְ אֵרַע, שֶׁכַּאֲשֶׁר שֵׁשׁ הַתַּלְמִידוֹת כְּבָר בָּגְרוּ וְהִגִּיעוּ לְפִרְקָן, הֵחֵלּוּ לְהִתְרַחֵשׁ אֵרוּעִים מְסֻיָּמִים. הַנְּסִיכָה דֵייזִי, בִּתּוֹ שֶׁל הַמֶּלֶךְ פוֹרְטוּנָטוּס, הַשַּׁלִּיט הַמְּכַהֵן, הָיְתָה הַיַּלְדָּה הַקְּטַנָּה הַיְּחִידָה בְּבֵית הַסֵּפֶר.

דֵייזִי כְּבָר שָׁהֲתָה בְּבֵית הַסֵּפֶר כְּשָׁנָה, כַּאֲשֶׁר יוֹם אֶחָד צִלְצֵל פַּעֲמוֹן הַדֶּלֶת, וְהַנַּעֲרָה הַמְּשָׁרֶתֶת הוֹפִיעָה בַּכִּתָּה כְּשֶׁכַּרְטִיס בִּקּוּר אָחוּז בִּקְצֵה סִינָרָהּ – כִּי יָדֶיהָ הָיוּ רְטֻבּוֹת מִשּׁוּם שֶׁזֶּה הָיָה יוֹם כְּבִיסָה.

"גֶ'נְטְלְמֶן אֶחָד הִגִּיעַ לִרְאוֹתֵךְ, הָעַלְמָה," אָמְרָה; וְהָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן הֻחְמְאָה לְמַדַּי, כִּי חָשְׁבָה שֶׁזֶּהוּ מֶלֶךְ מְכֻבָּד, שֶׁלּוֹ בַּת שֶׁרוֹצָה לִלְמֹד בְּבֵית הַסֵּפֶר.

אַךְ כְּשֶׁהִבִּיטָה בְּכַרְטִיס הַבִּקּוּר כְּבָר לֹא הִרְגִּישָׁה כֹּה מֻחְמֵאת וּמִלְמְלָה לְעַצְמָהּ בַּחֲשַׁאי "אוֹי לִי!" כִּי הִיא הִקְפִּידָה מְאוֹד עַל גִּנּוּנֵי הַנִּימוּסִין. לוּ הָיְתָה בּוֹטָה כְּמוֹ אֲחָדִים מֵאִתָּנוּ, הִיא הָיְתָה אוֹמֶרֶת "נוּדְנִיק!", וְלוּ הָיְתָה אַף בּוֹטָה יוֹתֵר מִכָּל אֶחָד מֵאִתָּנוּ, כָּךְ אֲנִי מְקַוָּה לְפָחוֹת, הִיא הָיְתָה אוֹמֶרֶת "שֶׁיֵּלֵךְ לְכָל הָרוּחוֹת!" הַכַּרְטִיס הָיָה גָּדוֹל וְזוֹהֵר וְעָלָיו אוֹתִיּוֹת מֻזְהָבוֹת. הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן קָרְאָה:

הָאַבִּיר דוֹלוֹרוֹ דֶה לָארָה

פְּרוֹפֵסוֹר לְקֶסֶם (לָבָן)

וּלְאָמָּנוּת שְׁחֹרָה.

מְלַמֵּד תַּלְמִידִים בִּמְקוֹם מְגוּרֵיהֶם.

לְלֹא הוֹצָאוֹת נִלְווֹת.

תְּנָאִים מְיֻחָדִים לְבָתֵּי סֵפֶר. מִשְׁתַּתֵּף בִּמְסִבּוֹת עֶרֶב.

הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן הֵנִיחָה אֶת סִפְרָהּ – הִיא מֵעוֹלָם לֹא לִמְּדָה בְּלִי סֵפֶר – הֶחְלִיקָה אֶת כּוֹבָעָהּ הַצָּהֹב וְאֶת תַּלְתַּלֶּיהָ הָאֲפֹרִים וְהָלְכָה לַטְּרַקְלִין הַקִּדְמִי כְּדֵי לִפְגּשׁ אֶת הָאוֹרֵחַ. הוּא קַד קִדָּה עֲמֻקָּה לְמַרְאָהּ. הוּא הָיָה גָּבוֹהַּ מְאוֹד וְרָעֵב לְמַרְאֶה, עִם עֵינַיִם שְׁחֹרוֹת וּפֶה בַּל-יְתֹאַר.

"לְעֹנֶג הוּא לִי," אָמַר וְחִיֵּךְ חִיּוּךְ שֶׁכְּמוֹ נוֹעַד לְהַצִּיג כָּל אַחַת מִשְּׁלוֹשִׁים וּשְׁתַּיִם שִׁנָּיו – מַעֲשֶׂה מְנֻמָּס מְאוֹד, אַךְ לֹא פָּשׁוּט כְּלָל – "לְעֹנֶג הוּא לִי לִפְגֹּש שׁוּב אֶת אַחַת מִתַּלְמִידוֹתַי מִשֶּׁכְּבָר."

"בְּטוּחַתְנִי שֶׁהָעֹנֶג הֲדָדִי," אָמְרָה הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן. אִם לְעִתִּים בִּלְתִּי אֶפְשָׁרִי לִהְיוֹת מְנֻמָּס וְדוֹבֵר אֱמֶת בְּעֵת וּבְעוֹנָה אַחַת, אֵין זֶה אַשְׁמָתִי אוֹ אַשְׁמַת הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן.

"טִיַּלְתִּי פֹּה וָשָׁם," אָמַר הַפְּרוֹפֵסוֹר, עֲדַיִן מְחַיֵּךְ לְאֵין שִׁעוּר, "וְהִרְחַבְתִּי אֶת מַאֲגַר הַחָכְמָה שֶׁלִּי. אָנוּ חַיִּים וְלוֹמְדִים, הֲלוֹא כֵן, עַלְמָתִי הַיְּקָרָה? וְכָעֵת אֲנִי מוֹרֶה מֻסְמָךְ הַרְבֵּה יוֹתֵר מִכְּפִי שֶׁהָיִיתִי כְּשֶׁהָיָה לִי הַכָּבוֹד לְלַמֵּד אוֹתָךְ. הַאִם אוּכַל לְקַוּוֹת לְמִשְׂרָה בְּבֵית הַסֵּפֶר שֶׁלָּךְ?"

"עֲדַיִן לֹא עָלָה בְּיָדִי לְהַסְדִּיר קוּרְס קֶסֶם," אָמְרָה מְנַהֶלֶת הַפְּנִימִיָּה; "זֶה תְּחוּם שֶׁהוֹרִים, וּבִמְיֻחָד הוֹרִים מַלְכוּתִיִּים, מְגַלִּים בּוֹ עִנְיָן מוּעָט בִּלְבַד."

"זֶה הָיָה הַמִּקְצוֹעַ הַמָּעֳדָף עָלַיִךְ," אָמַר הַפְּרוֹפֵסוֹר.

"כֵּן-אֲבָל-אֶה, אֵין סָפֵק שֶׁבְּיוֹם מִן הַיָּמִים -"

"אֲבָל אֲנִי זָקוּק לְמִשְׂרָה בִּדְחִיפוּת," אָמַר וְנִרְאָה מֻרְעָב מִתָּמִיד; "אֶלֶף לִירוֹת שְׂטֶרְלִינְג לִשְׁלוֹשָׁה-עָשָׂר שִׁעוּרִים – מְחִיר בִּמְיֻחָד בִּשְׁבִילֵךְ, עַלְמָתִי הַיְּקָרָה."

"זֶה בִּלְתִּי אֶפְשָׁרִי," אָמְרָה בְּרִשְׁמִיּוּת, וְהִיא דִּבְּרָה בְּתֹקֶף, כִּי יָדְעָה מִנִּסְיוֹן הֶעָבָר עַד כַּמָּה מְסֻכָּן לְהָנִיחַ לְקוֹסֵם לְהִתְקָרֵב לִנְסִיכָה. כִּמְעַט תָּמִיד נִגְרָם נֶזֶק כָּלְשֶׁהוּ.

"טוֹב וְיָפֶה," אָמַר הַפְּרוֹפֵסוֹר.

"עָלֶיךָ לְהָבִין," הִמְשִׁיכָה, "כָּל תַּלְמִידוֹתַי נְסִיכוֹת, וְהֵן אֵינָן נִדְרָשׁוֹת לְהִשְׁתַּמֵּשׁ בִּקְסָמִים, הֵן מְקַבְּלוֹת אֶת כָּל מָה שֶׁרַק יִרְצוּ בְּכָל מִקְרֶה."

"הַאִם אַתְּ אוֹמֶרֶת לִי לֹא?" אָמַר.

"אֲנִי אוֹמֶרֶת לְךָ לֹא וּמוֹדָה לְךָ," אָמְרָה.

וְאָז, לִפְנֵי שֶׁיָּכְלָה לַעֲצֹר בַּעֲדוֹ, הוּא זִנֵּק וְהָדַף אוֹתָהּ קַלּוֹת בְּדַרְכּוֹ אֶל הַדֶּלֶת, וְהִיא שָׁמְעָה אֶת מַגָּפָיו עַל הַשַּׁעֲוָנִית בַּמִּסְדְּרוֹן. הִיא דָּלְקָה בְּעִקְבוֹתָיו רַק כְּדֵי לִרְאוֹת אֶת דֶּלֶת הַכִּתָּה נִטְרֶקֶת וְנִנְעֶלֶת בְּפָנֶיהָ.

"לֹא הָיָה עוֹלֶה עַל דַּעְתִּי," אָמְרָה הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן. הִיא מִהֲרָה לְגַג הַבַּיִת וְיָרְדָה בַּאֲרֻבַּת הַכִּתָּה, שֶׁהִיא הִתְקִינָה בָּהּ מַדְרֵגוֹת לְמִקְרֶה שֶׁתִּפְרֹץ אֵשׁ אוֹ לִשְׁעַת חֵרוּם כָּלְשֶׁהִי. הִיא פָּסְעָה הַחוּצָה מִן הָאָח אֶל הַמַּרְבָד שֶׁבַּכִּתָּה, אַךְ עָשְׂתָה זֹאת שְׁנִיָּה אַחַת מְאֻחָר מִדַּי. שֶׁבַע הַנְּסִיכוֹת נֶעֶלְמוּ כֻּלָּן, וְהַפְּרוֹפֵסוֹר לְקֶסֶם עָמַד לְבַדּוֹ בֵּין הַשֻּׁלְחָנוֹת הַמֻּכְתָּמִים בִּדְיוֹ וְחִיֵּךְ אֶל הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן אֶת הַחִיּוּךְ הָרָחָב בְּיוֹתֵר שֶׁרָאֲתָה מֵעוֹדָהּ.

"הוֹ! אַתָּה אִישׁ נַכְלוּלִי, רַע, עַרְמוּמִי!" אָמְרָה וְנוֹפְפָה לְעֻמָּתוֹ בְּסַרְגֵּל בֵּית הַסֵּפֶר.

לְמָחֳרָת הָיְתָה שַׁבָּת, וּמֶלֶךְ הָאָרֶץ נִקְרָא כְּהֶרְגֵּלוֹ לָקַחַת אֶת בִּתּוֹ דֵייזִי לַחֲצִי יוֹם חֹפֶשׁ. הַמְּשָׁרֶתֶת שֶׁפָּתְחָה אֶת הַדֶּלֶת עָטְתָה סִינָר גַּס וְהָיוּ לָהּ גֶּחָלִים בַּשֵּׂעָר, וְהַמֶּלֶךְ חָשַׁב שֶׁהִיא לוֹבֶשֶׁת שַׂק וָאֵפֶר וְצִיֵּן זֹאת בְּעַצְבָּנוּת מְסֻיֶּמֶת, אֲבָל הַנַּעֲרָה אָמְרָה, "לֹא, אֲנִי רַק עוֹסֶקֶת בַּעֲבוֹדוֹת מִטְבָּח – אֲבָל כְּדַאי שֶׁתִּרְאֶה תָּ'עַלְמָה עַצְמָהּ."

וְאָז הוּבַל הַמֶּלֶךְ אֶל הַטְּרַקְלִין הַמְּפֹאָר בְּיוֹתֵר שֶׁהָיוּ בּוֹ פִּרְחֵי שַׁעֲוָה מְעֻצָּבִים בְּרֹב טַעַם, כִּסּוּיִים רְקוּמִים לְמִשְׁעֲנוֹת הַכִּסְּאוֹת וְאִיּוּרִים בְּצִבְעֵי מַיִם שֶׁנּוֹצְרוּ כֻּלָּם עַל יְדֵי הַתַּלְמִידוֹת, וּשְׁטִיחַ הַצֶּמֶר, שֶׁדּוֹדָתָהּ שֶׁל הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן, הַמְּרֻתֶּקֶת לְמִטָּתָהּ, אָרְגָה יָפֶה כָּל כָּךְ. טְרַקְלִין מְעֻדָּן מֵאֵין כָּמוֹהוּ, שֶׁנִּכַּר בּוֹ מַגַּע יָדָהּ הַמְּעֻדָּן שֶׁל אִשָּׁה.

הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן נִכְנְסָה פְּנִימָה בְּאִטִּיּוּת וּבְעֶצֶב. גְּלִימָתָהּ הָיְתָה אֲרוּגָה בִּקְפִידָה מִשַּׂק – וְקֻשְּׁטָה לְהַפְלִיא בְּגַחֲלֵי אֵשׁ קְטַנּוֹת תְּפוּרוֹת לְמִקְלַעַת מֻזְהֶבֶת – וְגֶחָלִים מְעַט גְּדוֹלוֹת יוֹתֵר הִתְנוֹדְדוּ עַל פְּתִילֵי מֶשִׁי שֶׁחֻבְּרוּ לְקַצְווֹת כּוֹבַע הַתַּחֲרָה שֶׁלָּהּ.

הַמֶּלֶךְ רָאָה מִיָּד שֶׁמַּשֶּׁהוּ מַטְרִיד אוֹתָהּ. "אֲנִי מְקַוֶּה שֶׁלֹּא הִקְדַּמְתִּי," אָמַר.

"כְּלָל וּכְלָל לֹא, הוֹד מַלְכוּתְךָ," עָנְתָה. "אַתָּה דַּיְקָן כְּתָמִיד, בְּדִיּוּק כְּפִי שֶׁנִּכְתַּב בְּהַמְלָצוֹתֶיךָ. קָרָה מַשֶּׁהוּ. וְלֹא אַחְמִיר אֶת צָרוֹתֶיךָ עַל יְדֵי כָּךְ שֶׁאֲנַסֶּה לְהַסְתִּיר זֹאת. בִּתֵּךְ דֵייזִי, גַּאֲוַת בֵּית סִפְרֵנוּ הַקָּטָן וְאוֹצָרֵנוּ הַיָּקָר, נֶעֶלְמָה. שֵׁשׁ חַבְרוֹתֶיהָ הַמַּלְכוּתִיּוֹת אִתָּהּ. לְעֵת עַתָּה הֵן אָמְנָם בְּטוּחוֹת, אֲבָל כָּרֶגַע אֵינֶנִּי יְכוֹלָה לְהָנִיחַ אֶת יָדִי עַל אַף לֹא אַחַת מֵהַשֶּׁבַע."

הַמֶּלֶךְ הִתְיַשֵּׁב בִּכְבֵדוּת עַל חֵלֶק מִמַּעֲרֶכֶת הַסַּפּוֹת הַנּוֹדַעַת הַמַּקְסִימָה בְּגוֹן הָאֱגוֹז (כֻּרְסָאוֹת לִגְבָרוֹת וְלַאֲדוֹנִים, סַפָּה וְשִׁשָּׁה כִּסְּאוֹת נוֹסָפִים) וְהִתְנַשֵּׁף בְּעֶצֶב. הוּא לֹא יָדַע מָה לוֹמַר. אַךְ מְנַהֶלֶת בֵּית הַסֵּפֶר כָּתְבָה לְעַצְמָהּ עַל לוּחַ צִפְחָה אֶת מָה שֶׁבְּכַוָּנָתָהּ לוֹמַר וְלָמְדָה זֹאת בְּעַל פֶּה, כָּךְ שֶׁהָיְתָה מְסֻגֶּלֶת לְדַבֵּר בְּשֶׁטֶף.

"הוֹד מַלְכוּתְךָ, אֵינֶנִּי הָאֲשֵׁמָה הַבִּלְעָדִית – תְּלֵה אוֹתִי אִם כֵּן – אוֹ, לְיֶתֶר דִּיּוּק, תְּלֵה אוֹתִי אִם עָלֶיךָ לַעֲשׂוֹת זֹאת; אַךְ תְּחִלָּה הַרְשֵׁה נָא לִי לְהַצִּיעַ לְךָ כַּמָּה מִלּוֹת הֶסְבֵּר."

וְהִיא הִתְיַשְּׁבָה וְסִפְּרָה לוֹ אֶת כָּל הַסִּפּוּר עַל בִּקּוּרוֹ שֶׁל הַפְּרוֹפֵסוֹר, וְעָצְרָה בְּדִיּוּק בַּמְּקוֹמוֹת שֶׁבָּהֶם עָצַרְתִּי אֲנִי כְּשֶׁסִּפַּרְתִּי לָכֶם אֶת אֲשֶׁר אֵרַע.

הַמֶּלֶךְ הִקְשִׁיב, מוֹרֵט בְּעַצְבָּנוּת אֶת שׁוּלָיו הַסְּגֻלִּים-אַרְגְּמָנִיִּים שֶׁל הַכִּסּוּי שֶׁעַל מִשְׁעֶנֶת הַכִּסֵּא.

"מֵעוֹלָם לֹא הִסְכַּנְתִּי עִם שִׁיטוֹתָיו שֶׁל הַפְּרוֹפֵסוֹר," אָמְרָה הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן בְּעֶצֶב; "וְתָמִיד הָיוּ לִי סְפֵקוֹת בְּנוֹגֵעַ לְיָשְׁרוֹ הַמּוּסָרִי, סְפֵקוֹת שֶׁהִתְבָּרְרוּ כָּעֵת כְּמֻצְדָּקִים. לְאַחַר שֶׁסִּיַּמְתִּי אֶת קוּרְס הַהוֹרָאָה אֶצְלוֹ לִפְנֵי כַּמָּה שָׁנִים, לָקַחְתִּי שׁוּרָה שֶׁל שִׁעוּרִים מִמּוֹרֶה מֻסְמָךְ מִמֶּנּוּ, וְהוֹדוֹת לְשִׁעוּרִים אֵלֶּה, שֶׁהָיוּ – אִם יֻרְשֶׁה לִי לוֹמַר – יְקָרִים לְהַחְרִיד, הִצְלַחְתִּי, לְמַרְבֵּה הַמַּזָּל, לִתְקֹעַ טְרִיז בַּגַּלְגַּל שֶׁל הַנּוֹכֵל חֲסַר הַמַּצְפּוּ -"

"הַאִם הִצַּלְתְּ אֶת הַנְּסִיכוֹת?" יִלֵּל הַמֶּלֶךְ.

"לֹא; אֲבָל אוּכַל לַעֲשׂוֹת זֹאת אִם הוֹד מַלְכוּתְךָ וְהַהוֹרִים הָאֲחֵרִים יַפְקִידוּ אֶת הָעִנְיָן כֻּלּוֹ בְּיָדַי."

"זֶה עִנְיָן חָמוּר בְּיוֹתֵר," אָמַר הַמֶּלֶךְ; "דֵייזִי הַקְּטַנָּה הַמִּסְכֵּנָה שֶׁלִּי -"

"בִּמְחִילָה מִכְּבוֹדְךָ," אָמְרָה מְנַהֶלֶת בֵּית הַסֵּפֶר, "אֲבַקֶּשְׁךָ לֹא לְיַחֵס חֲשִׁיבוּת רַבָּה מִדַּי לָאֵרוּעַ הַזֶּה. מוּבָן שֶׁזֶּה מְצַעֵר, אַךְ תְּאוּנוֹת לֹא נְעִימוֹת קוֹרוֹת בְּכָל בָּתֵּי הַסֵּפֶר, וְאֶת הַשְׁלָכוֹתֵיהֶן נִתָּן לִמְנֹעַ בְּאֶמְצָעוּת שִׁמּוּשׁ נָבוֹן בְּדֶרֶךְ אֶרֶץ וּבְשִׁקּוּל דַּעַת."

"אֶצְטָרֵךְ לִתְלוֹת אוֹתָךְ, אַתְּ יוֹדַעַת," אָמַר הַמֶּלֶךְ בְּסַפְקָנוּת.

"לְלֹא צֵל שֶׁל סָפֵק," אָמְרָה הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן, "וְאִם תַּעֲשֶׂה זֹאת, לְעוֹלָם לֹא תִּרְאֶה אֶת דֵייזִי שֶׁלְּךָ שׁוּב. חוֹבָתְךָ כְּהוֹרֶה – כֵּן – וְחוֹבָתְךָ כְּלַפַּי מִתְנַגְּשׁוֹת, וְזֶה בְּהֶחְלֵט עִנְיָן מַכְאִיב בְּיוֹתֵר."

"אֲבָל הַאִם אוּכַל לִבְטֹחַ בָּךְ?"

"עָלַי לְהַזְכִּיר לְךָ," אָמְרָה הָעַלְמָה וְזָקְפָה אֶת קוֹמָתָהּ כָּךְ שֶׁהַגֶּחָלִים נִרְעֲדוּ שׁוּב, "שֶׁהֶחְלַפְנוּ הַמְלָצוֹת מַשְׂבִּיעוֹת רָצוֹן בִּתְחִלַּת הַקֶּשֶׁר הָעִסְקִי בֵּינֵינוּ."

הַמֶּלֶךְ קָם. "טוֹב, הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן," אָמַר, "אֲנִי מְרֻצֶּה לַחֲלוּטִין מֵהִתְקַדְּמוּתָהּ שֶׁל דֵייזִי מִיּוֹם שֶׁהִיא נְתוּנָה לְהַשְׁגָּחָתֵךְ, וַאֲנִי חָשׁ שֶׁהַדָּבָר הַטּוֹב בְּיוֹתֵר יִהְיֶה לְהַפְקִיד אֶת הָעִנְיָן כֻּלּוֹ בְּיָדַיִךְ הָאֲמוּנוֹת."

מְנַהֶלֶת בֵּית הַסֵּפֶר הֶחֶוְתָה לוֹ קִדָּה, וְהוּא חָזַר לְאַרְמוֹן הַבְּדֹלַח שֶׁלּוֹ כְּמֶלֶךְ שְׁבוּר לֵב, כְּשֶׁכִּתְרוֹ נוֹטֶה בְּרִשּׁוּל עַל צַד אֶחָד שֶׁל רֹאשׁוֹ, וְאַפּוֹ הַיָּקָר הָאֻמְלָל אָדֹם מִבֶּכִי.

הַפְּנִימִיָּה הַמֻּבְחֶרֶת סָגְרָה אֶת שְׁעָרֶיהָ.

זְמַן חָלַף, אַךְ לֹא הִגִּיעוּ כָּל חֲדָשׁוֹת בְּנוֹגֵעַ לַנְּסִיכוֹת הָאֲבוּדוֹת.

הַמֶּלֶךְ מָצָא נֶחָמָה מוּעָטָה בָּעֻבְדָּה שֶׁחָסוּ עַל חַיָּיו שֶׁל יַלְדּוֹ הָאַחֵר, הַנָּסִיךְ דֵנִיס. דֵנִיס אָמְנָם הָיָה בְּסֵדֶר גָּמוּר וְיֶלֶד קָטָן וְנֶחְמָד בְּדַרְכּוֹ, אַךְ יֶלֶד אֵינוֹ יַלְדָּה.

הַמַּלְכָּה הָיְתָה שְׁבוּרַת לֵב אַף יוֹתֵר מִן הַמֶּלֶךְ, אַךְ הָיָה עָלֶיהָ לְהַשְׁגִּיחַ עַל עֲבוֹדוֹת הַבַּיִת וּלְהָכִין מַחְמַצִּים וְשִׁמּוּרִים וּלְהַטְלִיא אֶת גַּרְבָּיו שֶׁל הַנָּסִיךְ הַצָּעִיר, כָּךְ שֶׁלֹּא הָיָה לָהּ זְמַן רַב לְיַבֵּב, וְאַחֲרֵי שָׁנָה הִיא אָמְרָה לַמֶּלֶךְ –

"יַקִּירִי, אַתָּה חַיָּב לַעֲשׂוֹת מַשֶּׁהוּ כְּדֵי לְהַסִּיחַ אֶת דַּעְתְּךָ. אֵין זֶה הוֹלֵם שֶׁמֶּלֶךְ יֵשֵׁב וִייַבֵּב כָּל הַיּוֹם. עַכְשָׁו, עֲשֵׂה מַאֲמָץ; מְצָא מַשֶּׁהוּ שִׁמּוּשִׁי לַעֲשׂוֹת, גַּם אִם זֶה רַק פְּתִיחָה שֶׁל שׁוּק אוֹ הֲנָחַת אֶבֶן פִּנָּה."

"אֲנִי חוֹשֵׁשׁ מִשְּׁוָקִים," אָמַר הַמֶּלֶךְ; "הֵם כְּמוֹ דְּבוֹרִים – מְזַמְזְמִים וּמַדְאִיגִים; אַךְ הֲנָחַת אֶבֶן פִּנָּה -" וְאַחַר כָּךְ הוּא הִתְיַשֵּׁב וְחָשַׁב לִפְעָמִים, גַּם בְּלִי בֶּכִי, וְכָתַב לְעַצְמוֹ פְּתָקִים עַל גַּבֵּי מַעֲטָפוֹת מְשֻׁמָּשׁוֹת. כָּךְ שֶׁהַמַּלְכָּה הִרְגִּישָׁה שֶׁהִיא לֹא דִּבְּרָה לַשָּׁוְא. 

לְאַחַר חֹדֶשׁ נָשְׂאָה הַצָּעָתָהּ פְּרִי.

הַמֶּלֶךְ הִשִּׁיק חֲבֵרָה, וְ"רֵקְס פוֹרְטוּנָטוּס וְשׁוּת'" הָפְכָה כִּמְעַט מִיָּד לְחֶבְרַת הַבְּנִיָּה הַסְּפֵּקוּלַנְטִית הַגְּדוֹלָה בָּעוֹלָם.

אוּלַי אֵינְכֶם יוֹדְעִים מַהִי חֶבְרַת בְּנִיָּה סְפֵּקוּלַנְטִית. כְּדֵי שֶׁתֵּדְעוּ אֲסַפֵּר לָכֶם בְּמָה עָסְקוּ הַמֶּלֶךְ וְהַחֶבְרָה שֶׁהֵקִים.

הֵם קָנוּ אֶת כָּל הַיְּעָרוֹת וְהַשָּׂדוֹת הַיָּפִים שֶׁיָּכְלוּ לְהָנִיחַ יָדָם עֲלֵיהֶם וְחִלְּקוּ אוֹתָם לְרִבּוּעִים וְעָקְרוּ מֵהֶם אֶת כָּל הָעֵצִים וְהַצִּמְחִיָּה וּבִמְקוֹמָם סָלְלוּ רְחוֹבוֹת וְעַמּוּדֵי חַשְׁמַל וּבָתִּים כְּעוּרִים מִלְּבֵנִים צְהֻבּוֹת, בְּתִקְוָה שֶׁאֲנָשִׁים יִרְצוּ לָגוּר בָּהֶם. וּבְאֹפֶן מַתְמִיהַּ לְמַדַּי, הָאֲנָשִׁים אָכֵן בָּאוּ. וְכָךְ הַמֶּלֶךְ וְהַחֶבְרָה שֶׁהֵקִים הִרְוִיחוּ הַרְבֵּה מְאוֹד כֶּסֶף.

מְעַנְיֵן שֶׁכִּמְעַט כָּל הַהוֹן שֶׁבָּעוֹלָם נַעֲשָׂה בְּאֶמְצָעוּת הֲפִיכַת דְּבָרִים יָפִים לִדְבָרִים מְכֹעָרִים. יְצִירַת יֹפִי מִכִּעוּר הִיא מְלָאכָה שֶׁשְּׂכָרָהּ לֹא בְּצִדָּהּ.

הָרְחוֹבוֹת הַקְּטַנִּים הַכְּעוּרִים הִזְדַּחֲלוּ עוֹד וָעוֹד וְהִתְפַּשְּׁטוּ אֶל מִחוּץ לָעֲיָרָה, מְחַסְּלִים בְּדַרְכָּם אֶת הַשְּׁטָחִים הַיְּרֻקִּים כְּמוֹ זְחָלִים צְהֻבִּים וּמֻרְעָבִים, אַךְ כְּשֶׁהִגִּיעוּ לְרַגְלֵי גִּבְעַת הַתִּלְתָּן נֶאֶלְצוּ לַעֲצֹר. הַבְּעָלִים שֶׁל גִּבְעַת הַתִּלְתָּן לֹא הָיָה מוּכָן לִמְכֹּר אֶת אַדְמוֹתָיו בְּשׁוּם אֹפֶן וּבְשׁוּם מְחִיר שֶׁחֶבְרַת "רֵקְס פוֹרְטוּנָטוּס וְשׁוּת'" הָיְתָה יְכוֹלָה לְהַצִּיעַ. לַשָּׁוְא זִמְּנוּ מִשְׁפְּטָנֵי הַחֲבֵרָה אֶת מִשְׁפְּטָנֵי הַבְּעָלִים, לָבְשׁוּ אֶת גְּלִימוֹתֵיהֶם הַטּוֹבוֹת בְּיוֹתֵר, חָגְרוּ חֲרָבוֹת וּמָגִנִּים וְהִזְמִינוּ אוֹתָם לַאֲרוּחַת צָהֳרַיִם, הֶאֱכִילוּ וְהִשְׁקוּ אוֹתָם בּמַטְעַמִּים וּבְמַשְׁקָאוֹת מְשֻׁבָּחִים. גִּבְעַת הַתִּלְתָּן לֹא הָיְתָה לִמְכִירָה.

אַךְ לְבַסּוֹף נִקְרְאָה אִשָּׁה זְקֵנָה קְטַנָּה, לְבוּשָׁה כֻּלָּהּ בְּאָפֹר, אֶל מִשְׂרְדֵי הַמָּהָגוֹנִי וְהַפְּלִיז הַמְּמֹרָקִים שֶׁל הַחֶבְרָה לִפְגִישָׁה אִישִׁית עִם הַמֶּלֶךְ בִּכְבוֹדוֹ וּבְעַצְמוֹ.

"אֲנִי הַבְּעָלִים שֶׁל גִּבְעַת הַתִּלְתָּן," אָמְרָה, "וְאַתָּה יָכוֹל לִבְנוֹת עַל כָּל הַשֶּׁטַח, פְּרָט לְשִׁבְעַת הָאַקְרִים בְּרֹאשׁ הַגִּבְעָה וְלַחֲמִשָּׁה-עָשָׂר אַקְרִים נוֹסָפִים מִסְּבִיבָם, וְעָלֶיךָ לִבְנוֹת לִי חוֹמָה גְּבוֹהָה סְבִיב שִׁבְעַת הָאַקְרִים וְחוֹמָה נוֹסֶפֶת סְבִיב חֲמִשָּׁה-עָשָׂר הָאַקְרִים – וּשְׁתֵּיהֶן, תִּסְלַח לִי, מִלְּבֵנִים אֲדֻמּוֹת; לֹא מֵהַחֹמֶר הַזּוֹל הַצָּהֹב שֶׁאַתָּה בּוֹנֶה בּוֹ בְּדֶרֶךְ כְּלָל – וְאַתָּה חַיָּב לְחוֹקֵק חֹק חָדָשׁ שֶׁכָּל מִי שֶׁיִּגְנֹב מִפֵּרוֹתַי יִתָּלֶה עַל הָעֵץ שֶׁמִּמֶּנּוּ גָּנַב. זֶה הַכֹּל. מָה אַתָּה אוֹמֵר?"

הַמֶּלֶךְ אָמַר "כֵּן", כִּי מֵאָז אֲסוֹנוֹ לֹא הָיָה אִכְפַּת לוֹ מִדָּבָר, פְּרָט לִבְנִיָּה, וְנַפְשׁוֹ הַמַּלְכוּתִית הִשְׁתּוֹקְקָה לִרְאוֹת אֶת גִּבְעַת הַתִּלְתָּן הַיְּרֻקָּה נֶאֱכֶלֶת עַל יְדֵי זַחֲלֵי לְבֵנִים צְהֻבּוֹת בַּעֲלֵי גַּגּוֹת צִפְחָה. הוּא לֹא הִתְלַהֵב כְּלָל מֵהָרַעְיוֹן לִבְנוֹת שְׁתֵּי חוֹמוֹת מִלְּבֵנִים אֲדֻמּוֹת, אַךְ הוּא עָשָׂה זֹאת.

הָאִשָּׁה הַזְּקֵנָה רָצְתָה שֶׁהַחוֹמוֹת וְהָאַקְרִים יִהְיוּ מְסֻדָּרִים בַּצּוּרָה הַזֹּאת:

אֲבָל זוֹ הָיְתָה טִרְחָה גְּדוֹלָה בִּמְיֻחָד לְחַשֵּׁב אֶת שֶׁטַח הָאֲדָמָה שֶׁשְּׁנֵי הַמַּעְגָּלִים הָיוּ צְרִיכִים לְהָכִיל, עַד שֶׁכָּל הַמּוֹדְדִים מָרְטוּ אֶת שַׂעֲרוֹתֵיהֶם מְלוֹא הַחֹפֶן, וּבְסוֹפוֹ שֶׁל דָּבָר הַמֶּלֶךְ אָמַר, "לְכָל הָרוּחוֹת! עֲשׂוּ זֹאת כָּךְ," וְצִיֵּר לָהֶם תָּכְנִית מִצִּדּוֹ הַשֵּׁנִי שֶׁל אֶחָד מִדַּפֵּי הַחֻקִּים הַיְּשָׁנִים שֶׁל הַמֶּמְשָׁלָה. וְכָךְ עָשׂוּ, וְזֶה נִרְאָה כָּךְ –

הַגְּבֶרֶת הַזְּקֵנָה הָיְתָה מֻטְרֶדֶת מְאוֹד כְּשֶׁגִּלְּתָה שֶׁרַק חוֹמָה אַחַת מַפְרִידָה בֵּין הַבֻּסְתָּן שֶׁלָּהּ לְבֵין הָעוֹלָם, כְּפִי שֶׁאַתֶּם רוֹאִים בַּשִּׂרְטוּט בַּמָּקוֹם שֶׁבּוֹ רִבּוּעַ 1 וְרִבּוּעַ 15 נִפְגָּשִׁים; אַךְ הַמֶּלֶךְ אָמַר שֶׁאֵין בְּאֶפְשָׁרוּתוֹ לְמַמֵּן בְּנִיָּה מְחֻדֶּשֶׁת, וְשֶׁהִיא קִבְּלָה אֶת שְׁתֵּי הַחוֹמוֹת שֶׁבִּקְּשָׁה. כָּךְ שֶׁהָיָה עָלֶיהָ לְקַבֵּל זֹאת, אַךְ הִיא עָמְדָה עַל כָּךְ שֶׁהַמֶּלֶךְ יַשִּׂיג לָהּ כֶּלֶב בּוּלְדּוֹג פִּרְאִי שֶׁיָּעוּט עַל גְּרוֹנוֹ שֶׁל כָּל מִי שֶׁיַּעֲבֹר אֶת הַחוֹמָה בַּנְּקֻדָּה הַחַלָּשָׁה שֶׁבָּהּ רִבּוּעַ 1 וְרִבּוּעַ 15 נִפְגָּשִׁים. כָּךְ שֶׁהוּא הִשִּׂיג לָהּ כַּלְבַּת בּוּלְדּוֹג אֵימְתָנִית שֶׁשְּׁמָהּ מַרְתָה, רָתַם אוֹתָהּ לִרְצוּעָה מְשֻׁבֶּצֶת אַבְנֵי חֵן וְהֵבִיא לָהּ אוֹתָהּ בְּאֹפֶן אִישִׁי.

"מַרְתָה תָּעוּט עַל כָּל מִי שֶׁדָּם מַלְכוּתִי אֵינוֹ זוֹרֵם בְּעוֹרְקָיו," אָמַר. "מוּבָן שֶׁלֹּא תָּעֵז לִנְשֹׁךְ בֶּן מַלְכוּת; אִם כִּי בְּכָל מִקְרֶה אֲנָשִׁים שֶׁדָּם מַלְכוּתִי זוֹרֵם בְּעוֹרְקֵיהֶם לֹא שׁוֹדְדִים בֻּסְתָּנִים."

הָאִשָּׁה הַזְּקֵנָה נֶאֶלְצָה לְהַסְכִּים עִם דְּבָרָיו. הִיא קָשְׁרָה אֶת מַרְתָה לַפִּנָּה הַלֹּא מוּגֶנֶת שֶׁל הַמַּעְגָּל הַפְּנִימִי, שָׁתְלָה עֵץ תַּפּוּחַ קְטַנְטַן וּבָנְתָה בַּיִת וְהִתְיַשְּׁבָה בּוֹ וְחִכְּתָה.

וְהַמֶּלֶךְ הָיָה כִּמְעַט שָׂמֵחַ. הַזְּחָלִים הַצְּהֻבִּים הַגָּעֳלִיִּים חִסְּלוּ אֶת גִּבְעַת הַתִּלְתָּן כֻּלָּהּ – לְמַעֵט אֶת הַפִּסְגָּה הַיְּרֻקָּה, בַּמָּקוֹם שֶׁהָיוּ בּוֹ עֲצֵי הַתַּפּוּחַ וּשְׁתֵּי חוֹמוֹת הַלְּבֵנִים הָאֲדֻמּוֹת וְהַבַּיִת הַקָּטָן וְהָאִשָּׁה הַזְּקֵנָה.

הַמַּלְכָּה הַמִּסְכֵּנָה בָּדְקָה אֶת מַצַּב הָרִבּוֹת וְהַמַּחְמַצִּים וְהַשְּׂמִיכוֹת הַמִּתְכַּבְּסוֹת וְנִקְיוֹנוֹת הָאָבִיב, וּמִדֵּי פַּעַם גַּם אָמְרָה לְבַעֲלָהּ –

"פוֹרְטוּנָטוּס אֲהוּבִי, אַתָּה בֶּאֱמֶת חוֹשֵׁב שֶׁנִּתָּן לִבְטֹחַ בָּעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן? הַאִם אֵי-פַּעַם נִרְאֶה שׁוּב אֶת דֵייזִי שֶׁלָּנוּ?"

וְהַמֶּלֶךְ בָּחַשׁ בְּשַׂעֲרוֹתָיו הַחֲלָקוֹת עַד שֶׁהֵן נֶעֶמְדוּ כְּמוֹ מַקְּלוֹת גְּבִינָה מִתַּחַת לְכִתְרוֹ וְעָנָה –

"יַקִּירָתִי, אַתְּ חַיֶּבֶת לִהְיוֹת סַבְלָנִית; אַתְּ יוֹדַעַת שֶׁקִּבַּלְנוּ עָלֶיהָ הַמְלָצוֹת מְעֻלּוֹת לְלֹא רְבָב."

וְאָז, יוֹם אֶחָד, הִתְרַחֵשׁ אֵרוּעַ מְשַׂמֵּחַ בַּעֲיָרַת הַלְּבֵנִים הַצְּהֻבּוֹת הַחֲדָשָׁה שֶׁהַמֶּלֶךְ בָּנָה. שִׁשָּׁה נְסִיכִים נָאִים רְכוּבִים עַל גַּבֵּי סוּסִים לְבָנִים יְפֵהפִיִּים הִגִּיעוּ וּפִלְּסוּ אֶת דַּרְכָּם בָּרְחוֹבוֹת הַמְּאֻבָּקִים הַקְּטַנִּים. רִתְמוֹת הַמִּטְעָן הָרְקוּמוֹת שֶׁלָּהֶם בָּהֲקוּ מִכֶּסֶף וּמֵאַבְנֵי חֵן נוֹצְצוֹת מַבְרִיקוֹת, וְשִׁרְיוֹן הַזָּהָב שֶׁלָּהֶם הִזְדַּהֵר לְהַפְלִיא בְּאוֹר הַשֶּׁמֶשׁ, עַד שֶׁכָּל הַיְּלָדִים הַקְּטַנִּים מָחֲאוּ כַּפַּיִם, וּפְנֵיהֶם שֶׁל הַנְּסִיכִים הָיוּ צְעִירִים וּנְעִימִים וְנָאִים כָּל כָּךְ, שֶׁכָּל הַנָּשִׁים הַזְּקֵנוֹת אָמְרוּ, "יְבֹרַךְ לִבָּם הַיָּפֶה!"

מְיֻתָּר לְצַיֵּן שֶׁמּוּבָן מֵאֵלָיו שֶׁשֵּׁשֶׁת הַנְּסִיכִים חִפְּשׂוּ אַחַר שֵׁשׁ הַנְּסִיכוֹת שֶׁבָּגְרוּ וְהָיוּ מְאֻשָּׁרוֹת בַּפְּנִימִיָּה הַמֻּבְחֶרֶת לִבְנוֹת-מְלָכִים מְכֻבָּדִים. שֵׁשֶׁת אֲבוֹתֵיהֶם הַמַּלְכוּתִיִּים, שֶׁשָּׁהוּ שָׁנִים רַבּוֹת בְּמַסָּע בַּקָּצֶה הָאַחֵר שֶׁל הָעוֹלָם, טֶרֶם שָׁמְעוּ כִּי שֵׁשׁ הַנְּסִיכוֹת נֶעֶלְמוּ, וְלָכֵן נָתְנוּ לַנְּסִיכִים אֶת כְּתָבְתָּהּ שֶׁל הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן וְהוֹרוּ לָהֶם לְהִנָּשֵׂא לְשֵׁשׁ הַנְּסִיכוֹת לְלֹא דִּחוּי וּלְהָבִיאָן הַבַּיְתָה.

וְאוּלָם, כְּשֶׁהִגִּיעוּ אֶל הַפְּנִימִיָּה הַמֻּבְחֶרֶת לִבְנוֹת-מְלָכִים מְכֻבָּדִים, הָיָה הַבַּיִת סָגוּר, וְכַרְטִיס עַל הַחַלּוֹן הוֹדִיעַ שֶׁהַבַּיִת הַנֶּחְשָׁק הַזֶּה מֻצָּע לְהַשְׂכָּרָה בִּתְנָאִים נוֹחִים, מְרֹהָט אוֹ רֵיק. שַׁעֲוַת הַפֵּרוֹת תַּחַת הַצֵּל שֶׁהֵטִילָה הַזְּגוּגִית עֲדַיִן מָשְׁכָה אֶת הָעַיִן מִבַּעַד לַחַלּוֹנוֹת הַמְּאֻבָּקִים. שֵׁשֶׁת הַנְּסִיכִים הֵצִיצוּ כָּל אֶחָד בְּתוֹרוֹ מִבַּעַד לַחַלּוֹן. הֵם הֻקְסְמוּ מִן הָרִהוּט, וּמַגַּע יָדָהּ הַמְּעֻדָּן שֶׁל אִשָּׁה שֶׁנִּכַּר בַּכֹּל מָשַׁךְ אוֹתָם כְּמוֹ מַגְנֵט. הֵם שָׂכְרוּ אֶת הַבַּיִת, אַךְ אֶת אֲרוּחוֹתֵיהֶם סָעֲדוּ בָּאַרְמוֹן בְּהַזְמָנָה מְיֻחֶדֶת שֶׁל הַמֶּלֶךְ.

הַמֶּלֶךְ פוֹרְטוּנָטוּס סִפֵּר לַנְּסִיכִים אֶת הַסִּפּוּר הַנּוֹרָא עַל הֵעָלְמוּתָן שֶׁל כָּל בְּנוֹת בֵּית הַסֵּפֶר; וְכָל אֶחָד מִן הַנְּסִיכִים נִשְׁבַּע בִּכְבוֹדוֹ וּבְנִצַּב חַרְבּוֹ שֶׁיִּמְצָא אֶת הַנְּסִיכָה הַמְּיֹעֶדֶת לוֹ אוֹ יָמוּת בְּנִסָּיוֹן לְמוֹצְאָהּ. מוּבָן שֶׁכָּל נָסִיךְ הָיָה אָמוּר לָשֵׂאת נְסִיכָה אַחַת מְסֻיֶּמֶת, שֶׁשְּׁמָהּ צֻיַּן בַּהוֹרָאוֹת שֶׁקִּבֵּל, וְלֹא אַחֶרֶת, הַמְּיֹעֶדֶת לְאַחֵר.

הַלַּיְלָה הָרִאשׁוֹן שֶׁבּוֹ שָׁהוּ הַנְּסִיכִים בַּבַּיִת הַמְּרֹהָט עָבַר בְּשֶׁקֶט יַחֲסִי, וְכָךְ גַּם הַלַּיְלָה הַשֵּׁנִי וְהַלַּיְלָה הַשְּׁלִישִׁי וְהַלַּיְלָה הָרְבִיעִי, הַחֲמִישִׁי וְהַשִּׁשִּׁי, אֲבָל בַּלַּיְלָה הַשְּׁבִיעִי, כְּשֶׁהַנְּסִיכִים נֶעֶרְכוּ לְמִשְׂחַק דּוּקִים בַּכִּתָּה, הֵם שָׁמְעוּ לְפֶתַע פִּתְאוֹם קוֹל שֶׁלֹּא הָיָה שֶׁל אַף אֶחָד מֵהֶם. וְהַקּוֹל אָמַר, "תִּפְתְּחוּ אֶת אַפְרִיקָה."

הַנְּסִיכִים הִסְתַּכְּלוּ לְכָאן, לְשָׁם וּלְכָל עֵבֶר – אַךְ לֹא רָאוּ אִישׁ. הֵם לֹא לָמְדוּ דָּבָר עַל גִּלּוּי אֲרָצוֹת וְלֹא יָכְלוּ לִפְתֹּחַ אֶת אַפְרִיקָה גַּם לוּ רָצוּ.

"אוֹ תַּחְתְּכוּ דֶּרֶךְ מְצָרֵי פָּנָמָה," חָזַר הַקּוֹל וְאָמַר.

בְּמִקְרֶה, אִישׁ מִבֵּין שֵׁשֶׁת הַנְּסִיכִים לֹא הָיָה מְהַנְדֵּס. הֵם הוֹדוּ בְּכָךְ.

"אִם כֵּן, חִתְכוּ אֶת סִין!" אָמַר הַקּוֹל בְּיֵאוּשׁ.

"זֶה מַזְכִּיר אֶת רוּחַ הָרְפָאִים שֶׁהָעִתּוֹנִים כָּתְבוּ עָלֶיהָ לֹא מִזְּמַן," אָמַר אַחַד הַנְּסִיכִים.

וְאָז הֵם הֵבִינוּ פִּתְאוֹם שֶׁהַקּוֹל בּוֹקֵעַ מֵאֶחָד מִצֶּמֶד הַגְּלוֹבּוּסִים הַתְּלוּיִים בְּסוֹף הַכִּתָּה. זֶה הָיָה הַגְּלוֹבּוּס הָאַרְצִי.

"אֲנִי בִּפְנִים," אָמַר הַקּוֹל; "אֲנִי לֹא יָכוֹל לָצֵאת. אָנָּא מִכֶּם, חִתְכוּ אֶת הַגְּלוֹבּוּס – אֵיפֹה שֶׁרַק תּוּכְלוּ

 – וּתְנוּ לִי לָצֵאת. דֶּרֶךְ אַפְרִיקָה יִהְיֶה הֲכִי נוֹחַ."

הַנָּסִיךְ פְּרִימוּס פָּתַח אֶת אַפְרִיקָה בְּחַרְבּוֹ, וּמִתּוֹךְ הַגְּלוֹבּוּס הִתְגַּלְגֵּל הַחוּצָה חֶצְיוֹ הָעֶלְיוֹן שֶׁל הַפְּרוֹפֵסוֹר לְקֶסֶם.

"הַחֵצִי הַשֵּׁנִי שֶׁלִּי נִמְצָא שָׁם," הוּא אָמַר וְהִצְבִּיעַ עַל הַגְּלוֹבּוּס הַשְּׁמֵימִי. "תְּנוּ לָרַגְלַיִם שֶׁלִּי לָצֵאת בְּבַקָּ -"

אֲבָל הַנָּסִיךְ סֵקוּנְדוּס אָמַר, "לֹא כָּל כָּךְ מַהֵר," וְהַנָּסִיךְ טֶרְטְיוּס שָׁאַל, "לָמָּה סָגְרוּ אוֹתְךָ?"

"סָגְרוּ אוֹתִי רַק בִּגְלַל מְעַט קֶסֶם טְרַקְלִינִי שֶׁלֹּא רָאִיתָ כָּל יָמֶיךָ עַל פְּנֵי הָאֲדָמָה," אָמַר פֶּלֶג גּוּפוֹ הָעֶלְיוֹן שֶׁל הַפְּרוֹפֵסוֹר לְקֶסֶם.

"מָה, בֶּאֱמֶת?" אָמַר הַנָּסִיךְ קְוַּארְטוּס; "וּבְכֵן, הָרַגְלַיִם שֶׁלְּךָ עוֹד לֹא יוֹצְאוֹת הַחוּצָה. אֲנַחְנוּ זְקוּקִים לְעֶזְרָתוֹ שֶׁל קוֹסֵם מֻסְמָךְ. אַתָּה תַּתְאִים."

"אֲבָל לֹא כֻּלִּי כָּאן," אָמַר הַפְּרוֹפֵסוֹר.

"יֵשׁ מִמְּךָ דַּי וְהוֹתֵר," אָמַר הַנָּסִיךְ קְוִּינְטוּס.

"שִׂים לֵב עַכְשָׁו," אָמַר הַנָּסִיךְ סֵקְסְטוּס; "אֲנַחְנוּ רוֹצִים לִמְצֹא אֶת שֵׁשׁ הַנְּסִיכוֹת שֶׁלָּנוּ. יֵשׁ לָנוּ נִחוּשׁ לֹא רַע לְגַבֵּי אֹפֶן הֵעָלְמוּתָן; אֲבָל מָה שֶׁהָיָה הָיָה. תְּסַפֵּר לָנוּ אֵיךְ לִמְצֹא אוֹתָן, וּלְאַחַר הַחֲתֻנּוֹת שֶׁלָּנוּ נָשִׁיב אֶת רַגְלֶיךָ לְאוֹר הַיּוֹם."

"הַחֵצִי הַזֶּה שֶׁלִּי עוֹמֵד לְהִתְעַלֵּף," אָמַר חֲצִי הַפְּרוֹפֵסוֹר לְקֶסֶם.

"מָה אֲנַחְנוּ אֲמוּרִים לַעֲשׂוֹת?" אָמְרוּ הַנְּסִיכִים בְּנִימָה מְאַיֶּמֶת; "אִם לֹא תְּסַפֵּר לָנוּ, לְעוֹלָם לֹא תּוּכַל לַעֲמֹד עַל רַגְלֶיךָ."

"לְכוּ לִגְנֹב תַּפּוּחִים," אָמַר חֲצִי הַפְּרוֹפֵסוֹר בְּקוֹל צָרוּד וְהִתְעַלֵּף.

הֵם הִשְׁאִירוּ אוֹתוֹ לִשְׁכַּב עַל קַרְשֵׁי הָרִצְפָּה הַחֲשׂוּפִים בֵּין הַשֻּׁלְחָנוֹת הַמֻּכְתָּמִים בִּדְיוֹ וְיָצְאוּ מִיָּד לִגְנֹב תַּפּוּחִים. אֲבָל זֶה לֹא הָיָה כָּל כָּךְ פָּשׁוּט, מִכֵּיוָן שֶׁ"רֵקְס פוֹרְטוּנָטוּס וְשׁוּת'" בָּנְתָה וּבָנְתָה וּבָנְתָה, וְתַפּוּחִים לֹא גָּדְלוּ בְּחָפְשִׁיּוּת בַּאֲזוֹרִים אֵלֶּה שֶׁל הָאָרֶץ שֶׁ"נִפְתְּחוּ" עַל יְדֵי בּוֹנִים סְפֵּקוּלַנְטִים. 

אָז לְבַסּוֹף הֵם שָׁאֲלוּ אֶת הַנָּסִיךְ הַקָּטָן דֵנִיס לְאָן הוּא הוֹלֵךְ כְּשֶׁהוּא רוֹצֶה לְהַשִּׂיג תַּפּוּחִים. וְדֵנִיס אָמַר –

"הָאִשָּׁה הַזְּקֵנָה בְּפִסְגַּת גִּבְעַת הַתִּלְתָּן מְגַדֶּלֶת תַּפּוּחִים בְּשִׁבְעַת הָאַקְרִים שֶׁלָּהּ וּבַחֲמִשָּׁה-עָשָׂר הָאַקְרִים שֶׁלָּהּ, אֲבָל כֶּלֶב בּוּלְדּוֹג אֵימְתָנִי מַשְׁגִּיחַ עַל שִׁבְעַת הָאַקְרִים, וַאֲנִי גָּנַבְתִּי אֶת כָּל הַתַּפּוּחִים בַּחֲמִשָּׁה-עָשָׂר הָאַקְרִים בְּעַצְמִי."

"נְנַסֶּה אֶת שִׁבְעַת הָאַקְרִים," אָמְרוּ הַנְּסִיכִים.

"טוֹב מְאוֹד," אָמַר דֵנִיס; "יִתְלוּ אֶתְכֶם אִם תִּתָּפְסוּ. אָז מִשּׁוּם שֶׁאֲנִי סִבַּכְתִּי אֶתְכֶם, גַּם אֲנִי אָבוֹא, וְאִם לֹא תִּקְּחוּ אוֹתִי, אֲסַפֵּר לְכֻלָּם. אָז אֵין לָכֶם בְּרֵרָה!"

שֶׁכֵּן דֵנִיס הָיָה נָסִיךְ קָטָן וְהָגוּן בְּיוֹתֵר, וְהוּא לֹא הָיָה יָכוֹל לְהַרְשׁוֹת לְעַצְמוֹ לִשְׁלֹחַ אֲחֵרִים לְסַכָּנָה, אֶלָּא אִם כֵּן הוּא מִסְתַּכֵּן בְּעַצְמוֹ, וְהוּא לֹא הָיָה חוֹשֵׁב בִּכְלָל לִגְנֹב תַּפּוּחִים, אִלְמָלֵא הָיָה זֶה מְסֻכָּן לְפָחוֹת כְּמוֹ לְהוֹבִיל צָבָא לִקְרָב.

כָּךְ שֶׁהַנְּסִיכִים נֶאֶלְצוּ לְהַסְכִּים, וּבַלַּיְלָה הַבָּא שִׁלְשֵׁל דֵנִיס מֵחַלּוֹנוֹ חֶבֶל שֶׁהֵכִין מִכָּל הַגַּרְבַּיִם שֶׁאִמּוֹ סָרְגָה לוֹ, וְהַנְּסִיכִים הַבּוֹגְרִים חִכּוּ לוֹ מִתַּחַת לַחַלּוֹן, וְכֻלָּם פָּנוּ אֶל עֵבֶר הַבֻּסְתָּן שֶׁבְּפִסְגַּת גִּבְעַת הַתִּלְתָּן.

הֵם טִפְּסוּ עַל הַחוֹמָה בַּפִּנָּה הַמַּתְאִימָה, וּמַרְתָה, כַּלְבַּת הַבּוּלְדּוֹג, שֶׁהָיְתָה כַּלְבָּה מְיֻחֶסֶת וְיָדְעָה לְזַהוֹת נָסִיךְ כְּשֶׁנִּתְקְלָה בּוֹ, כִּשְׁכְּשָׁה בִּזְנָבָהּ הַקְּצַרְצַר בְּרֹב כָּבוֹד וְאִחֲלָה לָהֶם בְּהַצְלָחָה.

הַנְּסִיכִים הִתְגַּנְּבוּ עַל הָעֵשֶׂב הָרָטֹב מִטַּל שֶׁל הַבֻּסְתָּן וּבָדְקוּ עֵץ אַחֲרֵי עֵץ: לֹא הָיוּ כָּל תַּפּוּחִים עַל הָעֵצִים.

רַק לְבַסּוֹף, בְּלֵב לִבּוֹ שֶׁל הַבֻּסְתָּן, הָיָה עֵץ שֶׁלּוֹ גֶּזַע מִנְּחשֶׁת וְעַנְפֵי פְּלִיז וַעֲלֵי כֶּסֶף. וְעָלָיו הָיוּ תְּלוּיִים שִׁבְעָה תַּפּוּחֵי זָהָב יְפֵהפִיִּים.

אָז כָּל אֶחָד מֵהַנְּסִיכִים לָקַח לוֹ תַּפּוּחַ זָהָב אֶחָד, בְּשֶׁקֶט-בְּשֶׁקֶט, וְהֵם הָלְכוּ מִשָּׁם, מְמַהֲרִים לַחְזֹר אֶל חֲצִי הַפְּרוֹפֵסוֹר לְקֶסֶם, שֶׁיֹּאמַר לָהֶם מָה צָרִיךְ לַעֲשׂוֹת עַכְשָׁו. אִישׁ לֹא פִּקְפֵּק בְּכָךְ שֶׁהַפְּרוֹפֵסוֹר דָּבַר אֱמֶת; כִּי כְּשֶׁרַגְלֶיךָ תְּלוּיוֹת בַּאֲמִירַת אֱמֶת, לֹא תְּסַפֵּר מֵרְצוֹנְךָ דְּבַר שֶׁקֶר.

הֵם הִתְגַּנְּבוּ בְּשֶׁקֶט כְּכָל שֶׁיָּכְלוּ, כָּל אֶחָד מֵהֶם עִם תַּפּוּחַ זָהָב בְּיָדוֹ, אֲבָל אָז הַנָּסִיךְ דֵנִיס לֹא הָיָה יָכוֹל לְהִתְגַּבֵּר עַל הַפִּתּוּי לִנְעֹץ אֶת שִׁנָּיו בַּתַּפּוּחַ שֶׁלּוֹ. הוּא פָּתַח אֶת פִּיו לִרְוָחָה כְּדֵי לִנְגֹּס הֵיטֵב, אַךְ בָּרֶגַע הַבָּא הוּא יִלֵּל בְּקוֹל, כִּי הַתַּפּוּחַ הָיָה קָשֶׁה כְּאֶבֶן, וְהַיֶּלֶד הַמִּסְכֵּן שָׁבַר כִּמְעַט אֶת כָּל שִׁנָּיו הַקִּדְמִיּוֹת.

הוּא זָרַק אֶת הַתַּפּוּחַ הַצִּדָּה בְּזַעַם, וְכַעֲבֹר רֶגַע נֶחְפְּזָה הָאִשָּׁה הַזְּקֵנָה לָצֵאת מִבֵּיתָהּ. הִיא צָרְחָה. מַרְתָה נָבְחָה. הַנָּסִיךְ דֵנִיס יִלֵּל. כָּל הָעֲיָרָה הִתְעוֹרְרָה, וְשֵׁשֶׁת הַנְּסִיכִים נֶעֶצְרוּ וְנִלְקְחוּ תַּחַת מִשְׁמָר כָּבֵד לַמִּגְדָּל. אֶת דֵנִיס שִׁחְרְרוּ, הוֹדוֹת לְגִילוֹ הַצָּעִיר, מָה גַּם שֶׁהוּא אִבֵּד אֶת רֹב שִׁנָּיו, שֶׁזֶּה עֹנֶשׁ חָמוּר בִּמְיֻחָד, אֲפִלּוּ עַל גְּנֵבַת תַּפּוּחִים.

בְּבֹקֶר הַמָּחֳרָת יָשַׁב הַמֶּלֶךְ בְּהֵיכַל הַצֶּדֶק, וְהָאִשָּׁה הַזְּקֵנָה וְהַנְּסִיכִים הוּבְאוּ לְפָנָיו. אַחֲרֵי שֶׁסִּפְּרוּ אֶת שֶׁהִתְרַחֵשׁ, אָמַר –

"חֲבֵרִים יְקָרִים, אֲנִי מְקַוֶּה שֶׁתִּסְלְחוּ לִי – חֻקֵּי הַכְנָסַת הָאוֹרְחִים אָמְנָם נֻקְשִׁים – אֲבָל בַּעֲסָקִים כְּמוֹ בַּעֲסָקִים, אַחֲרֵי הַכֹּל. לֹא הָיִיתִי רוֹצֶה שֶׁתִּגָּרֵם כָּל אִי-נְעִימוּת חֻקָּתִית בְּשֶׁל עִנְיָן פָּעוּט כְּמוֹ זֶה; עָלַי לִתְלוֹת אֶת כֻּלְּכֶם מָחָר בַּבֹּקֶר."

הַנְּסִיכִים הִתְרַגְּזוּ עַד מְאוֹד, אֲבָל לֹא עוֹרְרוּ מְהוּמָה. הֵם בִּקְּשׁוּ לִרְאוֹת אֶת דֵנִיס וְאָמְרוּ לוֹ מָה עָלָיו לַעֲשׂוֹת.

וְכָךְ הָלַךְ לוֹ דֵנִיס אֶל הַבַּיִת הַמְּרֹהָט שֶׁפַּעַם הָיָה פְּנִימִיָּה מֻבְחֶרֶת לִבְנוֹת-מְלָכִים מְכֻבָּדִים. הַדֶּלֶת הָיְתָה נְעוּלָה, אֲבָל דֵנִיס יָדַע אֵיךְ לְהִכָּנֵס, כִּי אֲחוֹתוֹ גִּלְּתָה לוֹ פַּעַם, בְּאַחַד הַחַגִּים. הוּא עָלָה לַגַּג וְיָרַד בַּאֲרֻבַּת הַכִּתָּה.

שָׁם, עַל רִצְפַּת הַכִּתָּה, שָׁכַב חֲצִי הַפְּרוֹפֵסוֹר לְקֶסֶם, מִתְפַּתֵּל בְּכוֹחוֹתָיו הָאַחֲרוֹנִים וּמִצְטַוֵּחַ חֲלוּשׁוֹת בְּעֶצֶב.

"מָה עָלֵינוּ לַעֲשׂוֹת עַכְשָׁו?" שָׁאַל דֵנִיס.

"לְכוּ לִגְנֹב תַּפּוּחִים," לָחַשׁ חֲצִי הַפְּרוֹפֵסוֹר בְּקוֹל עַנּוֹת חֻלְשָׁה. "שַׁחְרֵר אֶת רַגְלַי. חֲתֹךְ אֶת הַדֻּבָּה הַגְּדוֹלָה – וְאוּלַי דֶּרֶךְ שְׁבִיל הֶחָלָב יִהְיֶה לָהֶן קַל לָצֵאת הַחוּצָה."

אֲבָל דֵנִיס יָדַע הֵיטֵב מָה הוּא אָמוּר לַעֲשׂוֹת.

"לֹא עַד שֶׁנִּמְצָא אֶת הַנְּסִיכוֹת הָאֲבוּדוֹת," אָמַר. "מָה עָלֵינוּ לַעֲשׂוֹת עַכְשָׁו?"

"לֵךְ לִגְנֹב תַּפּוּחִים, אֲנִי אוֹמֵר לְךָ," אָמַר חֲצִי הַפְּרוֹפֵסוֹר בְּכַעַס; "שִׁבְעָה תַּפּוּחִים – שֶׁבַע נְשִׁיקוֹת. קְטֹף אוֹתָם. לֵךְ לְךָ, יֶלֶד חֲסַר לֵב, הַשְׁאֵר אוֹתִי פֹּה לָמוּת. סִכּוֹת וּמַסְמְרִים בְּרַגְלַי."

דֵנִיס רָץ לְבֻּסְתָּן שִׁבְעַת הָאַקְרִים בְּפִסְגַּת גִּבְעַת הַתִּלְתָּן וְשָׁם מָצָא אֶת שֵׁשֶׁת הַנְּסִיכִים תְּלוּיִים עַל עֵץ הַתַּפּוּחַ. הַתַּלְיָן הָלַךְ הַבַּיְתָה לֶאֱכֹל אֲרוּחַת עֶרֶב, וְלֹא הָיָה אִישׁ בַּסְּבִיבָה. הַנְּסִיכִים לֹא מֵתוּ.

דֵנִיס טִפֵּס עַל הָעֵץ וְאֶת הַחֲבָלִים שֶׁעֲלֵיהֶם נִתְלוּ הַנְּסִיכִים חָתַךְ בְּאֶמְצָעוּת הָאוֹלָר שֶׁל בְּנוֹ שֶׁל הַגַּנָּן. (אוֹלָרִים מֻזְכָּרִים רַבּוֹת בִּסְפָרִים מְסֻיָּמִים, וְעַכְשָׁו גַּם אַתֶּם יְכוֹלִים לְהָבִין לָמָּה כָּל כָּךְ מִתְלַהֲבִים מֵהֶם).

הַנְּסִיכִים נָפְלוּ לַקַּרְקַע, וּכְשֶׁשָּׁבוּ לְעֶשְׁתּוֹנוֹתֵיהֶם סִפֵּר לָהֶם דֵנִיס מָה עָשָׂה.

"הוֹ! מַדּוּעַ הִפַּלְתָּ אוֹתָנוּ?" שָׁאֲלוּ הַנְּסִיכִים, "חָלַמְנוּ חֲלוֹמוֹת כָּל כָּךְ שְׂמֵחִים."

"וּבְכֵן," אָמַר דֵנִיס וְסָגַר בְּאַחַת אֶת הָאוֹלָר שֶׁל בְּנוֹ שֶׁל הַגַּנָּן, "לָזֶה יִקָּרֵא כְּפִיּוּת טוֹבָה!" וְהִפְנָה אֶת גַּבּוֹ וְהָלַךְ מִשָּׁם. אֲבָל הֵם רָצוּ אַחֲרָיו בִּמְהִירוּת וְהוֹדוּ לוֹ וְהִסְבִּירוּ לוֹ שֶׁהֵם חָלְמוּ עַל טִיּוּלִים יָד בְּיָד עִם הַנְּסִיכוֹת הַיְּקָרוֹת וְהַמַּקְסִימוֹת בְּיוֹתֵר בָּעוֹלָם כֻּלּוֹ.

"וּבְכֵן", אָמַר דֵנִיס, "אֵין בָּזֶה שׁוּם טַעַם. עֲלֵיכֶם לִמְצֹא נְסִיכוֹת מֻסְמָכוֹת מִשֶּׁלָּכֶם, וַחֲצִי הַפְּרוֹפֵסוֹר אוֹמֵר, 'לְכוּ לִגְנֹב תַּפּוּחִים.'"

"אֲבָל כְּבָר לֹא נִשְׁאֲרוּ תַּפּוּחִים לִגְנֹב," אָמְרוּ הַנְּסִיכִים – וְאוּלָם, כְּשֶׁהִבִּיטוּ בָּעֵץ רָאוּ שֶׁתַּפּוּחֵי הַזָּהָב חָזְרוּ אֵלָיו כְּבֶעָבָר.

וְכָךְ כָּל אֶחָד מֵהֶם שָׁב וּבָחַר לוֹ תַּפּוּחַ. דֵנִיס בָּחַר הַפַּעַם תַּפּוּחַ שׁוֹנֶה. הוּא קִוָּה שֶׁהַפַּעַם יִהְיֶה רַךְ יוֹתֵר. בַּפַּעַם הָאַחֲרוֹנָה הוּא בָּחַר בַּתַּפּוּחַ הַגָּדוֹל בְּיוֹתֵר – אֲבָל עַכְשָׁו קָטַף אֶת הַתַּפּוּחַ הַקָּטָן בְּיוֹתֵר.

"שֶׁבַע נְשִׁיקוֹת!" הוּא קָרָא וְהֵחֵל לְנַשֵּׁק אֶת תַּפּוּחַ הַזָּהָב הַקָּטָן.

כָּל אֶחָד מֵהַנְּסִיכִים נִשֵּׁק אֶת הַתַּפּוּחַ שֶׁבְּיָדוֹ, עַד שֶׁקּוֹלָן שֶׁל הַנְּשִׁיקוֹת נִשְׁמַע כִּלְחִישַׁת רוּחַ הָעֶרֶב בְּעַלְוַת הָעֵצִים. וּמוּבָן שֶׁבַּנְּשִׁיקָה הַשְּׁבִיעִית גִּלָּה כָּל אֶחָד מִן הַנְּסִיכִים שֶׁאֵין הוּא מַחְזִיק בְּיָדוֹ תַּפּוּחַ, אֶלָּא אֶת אֶצְבְּעוֹת יָדָהּ שֶׁל נְסִיכָה חִנָּנִית. וּבַאֲשֶׁר לְדֵנִיס, הוּא קִבֵּל אֶת אֲחוֹתוֹ הַקְּטַנָּה דֵייזִי, וְהוּא שָׂמַח כָּל כָּךְ, שֶׁמִּיָּד הִבְטִיחַ שֶׁיִּתֵּן לָהּ אֶת הַשַּׁרְקָנִים שֶׁלּוֹ וְאֶת הָאֹסֶף הַשָּׁלֵם שֶׁל בּוּלֵי הַדֹּאַר מֵחוּ"ל.

"מָה שְׁמֵךְ, גְּבֶרֶת יְקָרָה וּמַקְסִימָה?" שָׁאַל הַנָּסִיךְ פְּרִימוּס.

"סֵקְסְטָה," אָמְרָה הַנְּסִיכָה. וְאָז הִתְבָּרֵר שֶׁכָּל אֶחָד מִן הַנְּסִיכִים בָּחַר בַּתַּפּוּחַ הַלֹּא נָכוֹן, כָּךְ שֶׁהַנְּסִיכָה שֶׁאָחַז בְּיָדָהּ לֹא הָיְתָה זוֹ שֶׁצֻּיְּנָה בַּהוֹרָאוֹת. סֵקוּנְדוּס קָטַף אֶת הַתַּפּוּחַ שֶׁהֶחְזִיק אֶת קְוִּינְטָה, וְטֶרְטְיוּס אָחַז בִּקְוַּארְטָה וְכָךְ הָלְאָה – וְכַצָּפוּי הַכֹּל הִסְתַּבֵּךְ לְגַמְרֵי.

וְעִם זֹאת אִישׁ לֹא רָצָה לְשַׁנּוֹת אֶת בְּחִירָתוֹ.

וְאָז יָצְאָה מִבֵּיתָהּ הָאִשָּׁה הַזְּקֵנָה וְהִסְתַּכְּלָה בָּהֶם וְצִחְקְקָה וְאָמְרָה –

"עֲלֵיכֶם לִשְׂמֹחַ בְּחֶלְקְכֶם."

"אֲנַחְנוּ שְׂמֵחִים בְּחֶלְקֵנוּ," אָמְרוּ כָּל הַשְּׁנֵים-עָשָׂר, "אֲבָל מָה בְּנוֹגֵעַ לַהוֹרִים שֶׁלָּנוּ?"

"הֵם יִצְטָרְכוּ לְהַשְׁלִים עִם הַבְּחִירָה שֶׁלָּכֶם," אָמְרָה הָאִשָּׁה הַזְּקֵנָה, "זֶה מַשֶּׁהוּ שֶׁכָּל הַהוֹרִים חַיָּבִים לִלְמֹד לַעֲשׂוֹת."

"אֲנִי חוֹשֵׁב שֶׁאַתֶּם צְרִיכִים לִלְמֹד אֵיךְ לִבְחֹר כְּמוֹ שֶׁצָּרִיךְ," אָמַר דֵנִיס; "אֲנִי הַיָּחִיד שֶׁקִּבֵּל אֶת הַנְּסִיכָה הַנְּכוֹנָה – כִּי לֹא הָיִיתִי חַמְדָן; לָקַחְתִּי אֶת הַתַּפּוּחַ הַקָּטָן בְּיוֹתֵר."

הַנְּסִיכָה הַגְּבוֹהָה בְּיוֹתֵר הֶרְאֲתָה לוֹ סִימָן אָדֹם עַל זְרוֹעָהּ, בַּמָּקוֹם שֶׁבּוֹ נָגְסוּ שִׁנָּיו הַקְּטַנּוֹת שְׁנֵי לֵילוֹת קֹדֶם לָכֵן, וְכֻלָּם צָחֲקוּ.

אַךְ הָאִשָּׁה הַזְּקֵנָה אָמְרָה –

"הֵם לֹא יְכוֹלִים לְהַחְלִיף, יַקִּירִי. כְּשֶׁנָּסִיךְ בּוֹחֵר תַּפּוּחַ זָהָב שֶׁיֵּשׁ נְסִיכָה בְּתוֹכוֹ וְנוֹשֵׁק לוֹ עַד שֶׁהִיא יוֹצֵאת, הוּא לְעוֹלָם לֹא יַחְשֹׁק בִּנְסִיכָה אַחֶרֶת, וְגַם הִיא לְעוֹלָם לֹא תַּחְשֹׁק בְּנָסִיךְ אַחֵר."

בְּעוֹדָהּ מְדַבֶּרֶת הָלְכָה הָאִשָּׁה הַזְּקֵנָה וְהִצְעִירָה עוֹד וָעוֹד וָעוֹד, וּכְשֶׁסִּיְּמָה אֶת דְּבָרֶיהָ כְּבָר הָיְתָה צְעִירָה לְמַדַּי, בַּת חֲמִשִּׁים וְחָמֵשׁ לְכָל הַיּוֹתֵר. וְזוֹ הָיְתָה הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן!

תַּלְמִידוֹתֶיהָ פָּסְעוּ קָדִימָה בָּזוֹ אַחַר זוֹ וְקַדּוּ לָהּ, וְהִיא הִרְשְׁתָה לָהֶן לְנַשֵּׁק אוֹתָהּ עַל לֶחְיָהּ, מַמָּשׁ כְּאִלּוּ הִגִּיעַ זְמַנָּן לְהִפָּרֵד.

וְאָז, כֻּלָּם יַחַד, וּבְשִׂמְחָה גְּדוֹלָה, יָרְדוּ אֶל הַבַּיִת הַמְּרֹהָט שֶׁהָיָה פַּעַם פְּנִימִיָּה מֻבְחֶרֶת, וּכְשֶׁחֲצִי הַפְּרוֹפֵסוֹר נָתַן אֶת דִּבְרָתוֹ שֶׁיְּוַתֵּר עַל הַקֶּסֶם וְיִבְחַר בְּמִקְצוֹעַ מְכֻבָּד, הֵם הוֹצִיאוּ אֶת רַגְלָיו מִן הַגְּלוֹבּוּס הַשְּׁמֵימִי וְהֶחְזִירוּ לוֹ אוֹתָן; וְהוּא חִבֵּר אֶת שְׁנֵי חֲצָאָיו וְהָלַךְ מִשָּׁם, נָחוּשׁ לִחְיוֹת וְלָמוּת כִּשְׁרַבְרָב הָגוּן.

"כִּשְׁרוֹנוֹתַי לֹא יְבֻזְבְּזוּ לָרִיק," אָמַר; "מְעַט אֲחִיזַת עֵינַיִם יְעִילָה מְאוֹד בְּרֹב הַמִּקְצוֹעוֹת."

כְּשֶׁהַמֶּלֶךְ בִּקֵּשׁ מֵהָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן לִנְקֹב בַּפְּרָס הַמַּגִּיעַ לָהּ עַל הֲשָׁבַת הַנְּסִיכוֹת, הִיא אָמְרָה –

"הָשֵׁב אֶת הַשְּׁטָחִים הַיְּרֻקִּים, הוֹד מַלְכוּתְךָ."

וְכָךְ חֶבְרַת "רֵקְס פוֹרְטוּנָטוּס וְשׁוּת'" הִקְדִּישָׁה אֶת עַצְמָהּ לַהֲרִיסָה וְלַפִּנּוּי שֶׁל הָרְחוֹבוֹת הַצְּהֻבִּים שֶׁבָּנְתָה. וְכַיּוֹם הָאָרֶץ יְרֻקָּה וְיָפָה כִּמְעַט כְּמוֹ שֶׁהָיְתָה לִפְנֵי שֶׁהַנְּסִיכָה דֵייזִי וְשֵׁשׁ נְסִיכוֹת הַפְּנִימִיָּה הָפְכוּ לְתַפּוּחֵי זָהָב.

"זֶה הָיָה חָכָם מְאוֹד מִצִּדָּהּ שֶׁל הָעַלְמָה פִיצְרוֹי רוֹבִּינְסוֹן לִסְגֹּר אֶת הַפְּרוֹפֵסוֹר בִּשְׁנֵי הַגְּלוֹבּוּסִים," אָמְרָה הַמַּלְכָּה; "לְלֹא סָפֵק, יֵשׁ יִתְרוֹן לִלְקִיחַת שִׁעוּרִים פְּרָטִיִּים מִפִּי מוֹרִים מֻמְחִים בִּתְחוּמָם."

"נָכוֹן," אָמַר הַמֶּלֶךְ, "תָּמִיד אָמַרְתִּי שֶׁאִי אֶפְשָׁר לִטְעוֹת אִם מַקְפִּידִים עַל הַמְלָצוֹת מְעֻלּוֹת לְלֹא רְבָב."
